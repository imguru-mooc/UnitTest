        {
            if (string.IsNullOrEmpty(input))
            {
                return input;

            Array.Reverse(charArray);

        public bool IsPalindrome(string input)
        {
            if (string.IsNullOrEmpty(input))
            {
                return true;
            }

            string reversed = Reverse(cleaned);

        public int CountWords(string input)

            {
                return 0;
